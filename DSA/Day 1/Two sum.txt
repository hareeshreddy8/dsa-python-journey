class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        n = len(nums)
        hashmap = {}
        for i in range(n):
            compliment = target - nums[i]

            if compliment in hashmap:
                return[hashmap[compliment],i]
            
            hashmap[nums[i]] = i

#complexity O(n)
